import { FormattedMessage,FormattedDate } from "react-intl";
    import { DataTable } from "grommet";
    export default function CustomerTable({ customers }) {
        const columns = [
            { property: "avatarUrl", header: <FormattedMessage id="Customer" defaultMessage="avatarUrl"/>, render: val => (val.avatarUrl) },
            { property: "createdAt", header: <FormattedMessage id="Customer" defaultMessage="createdAt"/>, render: val => (<FormattedDate value={val.createdAt}/>) },
            { property: "email", header: <FormattedMessage id="Customer" defaultMessage="email"/>, render: val => (val.email) },
            { property: "id", header: <FormattedMessage id="Customer" defaultMessage="id"/>, render: val => (val.id) },
            { property: "name", header: <FormattedMessage id="Customer" defaultMessage="name"/>, render: val => (val.name) },
            { property: "phone", header: <FormattedMessage id="Customer" defaultMessage="phone"/>, render: val => (val.phone) },
            { property: "updatedAt", header: <FormattedMessage id="Customer" defaultMessage="updatedAt"/>, render: val => (<FormattedDate value={val.updatedAt}/>) }
        ];
        return (<DataTable columns={columns} data={customers}/>);
    }
